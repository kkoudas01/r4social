# 1. Υπολογισμός των 5 σημείων ελέγχου (Quantiles)
# Αυτά είναι τα σημεία στα οποία θέλεις να αντιστοιχίσεις συγκεκριμένα χρώματα.
quantiles <- quantile(sf_final$ΨΗΦΙΑΚΕΣ_ΥΠΗΡΕΣΙΕΣ, 
                      probs = c(0, 0.25, 0.50, 0.75, 1), 
                      na.rm = TRUE)

# 2. Ορισμός των 5 χρωμάτων
# Επέλεξε 5 διαφορετικά χρώματα για τα 5 σημεία ελέγχου.
my_5_colors <- c("red", "orange", "yellow", "lightgreen", "darkgreen")

# 3. Ορισμός των θέσεων (values)
# Καθώς το scale_fill_gradientn λειτουργεί με τιμές από 0 έως 1,
# πρέπει να μετατρέψουμε τις τιμές των ποσοστημορίων σε κλίμακα 0-1
# (δηλαδή, να τις κανονικοποιήσουμε).
min_val <- min(sf_final$ΨΗΦΙΑΚΕΣ_ΥΠΗΡΕΣΙΕΣ, na.rm = TRUE)
max_val <- max(sf_final$ΨΗΦΙΑΚΕΣ_ΥΠΗΡΕΣΙΕΣ, na.rm = TRUE)

# Υπολογισμός κανονικοποιημένων θέσεων (τιμές από 0 έως 1)
# Η φόρμουλα είναι: (τιμή - min) / (max - min)
normalized_values <- (quantiles - min_val) / (max_val - min_val)

# Βήμα 3.2: Δημιουργία Χρωροπληθούς Χάρτη
choropleth_map <- ggplot(data = sf_final) +
  geom_sf(aes(fill = ΨΗΦΙΑΚΕΣ_ΥΠΗΡΕΣΙΕΣ),  
          color = "gray70",  
          linewidth = 0.001) +
  
  # === Η ΤΕΛΙΚΗ ΛΥΣΗ ΕΙΝΑΙ ΕΔΩ! ===
  scale_fill_gradientn(
    colors = my_5_colors, # Τα 5 χρώματα που ορίσαμε
    values = normalized_values, # Οι 5 κανονικοποιημένες θέσεις (0 έως 1)
    
    name = "Ψηφιακές Υπηρεσίες Δήμου",
    na.value = "white"
  ) +
  # =================================

  theme_void() +
  
  labs(
    title = "Βαθμολογία Υπηρεσιών Δήμων (2025)",
    caption = "Δεδομένα: Υπουργείο Εσωτερικών - Ελληνική Στατιστική Αρχή"
  ) +
  
  theme(
    plot.title = element_text(hjust = 0.5, size = 16, face = "bold"),
    plot.subtitle = element_text(hjust = 0.5, size = 12),
    legend.position = "right"
  )

# Προβολή του χάρτη
print(choropleth_map)
