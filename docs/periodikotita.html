  n <- length(listPeriod)

  seira <- paste(c("I(sin(2*pi*t/",T,"))+I(cos(2*pi*t/",T,"))"), collapse="")
  if (n>1) {
    for (i in 2:n) {
    T <- listPeriod[i]

    seira <- paste(c(seira,orosN), collapse="+")
  }
  }
  formula_string <- paste("dedomena ~", seira)
  lm_model <- lm(as.formula(formula_string))
  lm_model
}
